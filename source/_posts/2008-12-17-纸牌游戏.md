void pokergame()  
{  
int i,j;  
//....................................................  
//选出不同的27个随机数

for(i=0;i<27;i++)  
{ j=0;  
randpoker[i]=rand()%54;  
while(i!=0&&(randpoker[i]!=randpoker[j])&&(j<i))  
{  
j++;  
}  
  
if (i!=0&&j!=i) --i;  
}  
}
